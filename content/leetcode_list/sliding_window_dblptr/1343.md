+ 定長度的 sliding window，秒殺
```cpp
class Solution {
public:
    int numOfSubarrays(vector<int>& arr, int k, int threshold) {
        threshold *= k;
        int sum = 0;
        for (int i = 0; i < k; i++) {
            sum += arr[i];
        }
        int res = 0;
        if (sum >= threshold) res++;
        for (int i = k; i < arr.size(); i++) {
            sum += (arr[i] - arr[i-k]);
            if (sum >= threshold) res++;
        }
        return res;
    }
};
```
