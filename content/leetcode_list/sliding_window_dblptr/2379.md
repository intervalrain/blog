+ 定長度的 sliding window，秒殺
```cpp
class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int cnt = 0;
        for (int i = 0; i < k; i++) {
            if (blocks[i] == 'B') cnt++;
        }
        int res = k-cnt;
        for (int i = k; i < blocks.size(); i++) {
            if (blocks[i] == 'B') cnt++;
            if (blocks[i-k] == 'B') cnt--;
            res = min(res, k-cnt);
        }
        return res;
    }
};
```
